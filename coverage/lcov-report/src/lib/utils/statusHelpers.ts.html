// ステータスの日本語ラベル
export const getStatusLabel = (status: string): string => {
  switch (status) {
    case 'NEW':
      return '新規';
    case 'IN_PROGRESS':
      return '回答中';
    case 'PENDING_APPROVAL':
      return '確認待ち';
    case 'CLOSED':
      return '完了';
    default:
      return status;
  }
};
